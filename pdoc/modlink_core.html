 34    def __init__(self, path: Path | None = None, parent: QObject | None = None) -> None:
 35        super().__init__(parent=parent)
 36        existing = type(self)._instance
 37        if existing is not None and existing is not self:
 38            raise RuntimeError(

 45    def get(self, key: str, default: Any = None) -> Any:
 46        current = self._settings
 47        for part in self._parts(key):
 48            if not isinstance(current, dict) or part not in current:
 49                return default
 50            current = current[part]
 51        return current
 52
 53    def set(self, key: str, value: Any, *, persist: bool = True) -> None:
 54        parts = self._parts(key)
 55        current = self._settings
 56        for part in parts[:-1]:
 57            nested = current.get(part)
 58            if not isinstance(nested, dict):
 59                nested = {}
 60                current[part] = nested
 61            current = nested
 62        current[parts[-1]] = value
 63        self.sig_setting_changed.emit(
 64            SettingChangedEvent(key=key, value=value, ts=time.time())
 65        )
 66        if persist:
 67            self.save()
 68
 69    def remove(self, key: str, *, persist: bool = True) -> None:
 70        parts = self._parts(key)
 71        current = self._settings
 72        for part in parts[:-1]:
 73            nested = current.get(part)
 74            if not isinstance(nested, dict):
 75                return
 76            current = nested
 77        if parts[-1] not in current:
 78            return
 79        current.pop(parts[-1], None)
 80        self.sig_setting_changed.emit(
 81            SettingChangedEvent(key=key, value=None, ts=time.time())
 82        )
 83        if persist:
 84            self.save()
 85
 86    def snapshot(self) -> dict[str, Any]:
 87        return json.loads(json.dumps(self._settings, ensure_ascii=False))
 88
 89    def save(self) -> None:
 90        self._path.parent.mkdir(parents=True, exist_ok=True)
 91        self._path.write_text(

 94        )

 97    def _read_payload(self) -> dict[str, Any]:
 98        try:
 99            if self._path.exists():
100                payload = json.loads(self._path.read_text(encoding="utf-8"))
101                if isinstance(payload, dict):

103        except (OSError, json.JSONDecodeError):
