160        while True:
161            try:
162                if self._drain_commands():
163                    return

416    def _submit_command(
417        self,
418        worker_method: Callable[..., None],
419        *args: object,
420    ) -> Future[None]:
421        future: Future[None] = Future()
422        if not self.is_started:
423            future.set_exception(RuntimeError("ACQ_NOT_STARTED"))
424            return future
425        if not self._accepting_commands:
426            future.set_exception(RuntimeError("ACQ_SHUTTING_DOWN"))
427            return future
428
429        def _run_command() -> None:
430            try:
431                worker_method(*args)
432            except Exception as exc:
433                if not future.done():
434                    future.set_exception(exc)
435                return
436            if not future.done():
437                future.set_result(None)
438
439        self._command_queue.put((_run_command, future))
440        return future

442    def _fail_pending_commands(self, message: str) -> None:
443        while True:
444            try:
445                item = self._command_queue.get_nowait()
446            except queue.Empty:
447                return
448            if item is self._shutdown_sentinel:
449                continue
450            _command, future = item
451            if future is None or future.done():
452                continue
453            future.set_exception(RuntimeError(message))

455    def _finalize_shutdown(self) -> None:
456        self._set_state("idle")
457        self._started = False
458        self._accepting_commands = True
459        self._thread = None
460        if self._frame_stream is not None:
461            self._frame_stream.close()
462            self._frame_stream = None

464    def _take_worker_exit_error(self) -> Exception | None:
465        with self._lock:
466            error = self._worker_exit_error
467            self._worker_exit_error = None
